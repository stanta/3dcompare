111 void STLViewer::compareLastMeshes()
112 {
113  QList<QMdiSubWindow *> subList = mdiArea->subWindowList();
114  int subSize = subList.size();
115  if (subSize > 1) {
116  statusBar()->showMessage(tr("Compare last two meshes"), 2000);
117 
118  GLMdiChild *child1 = qobject_cast<GLMdiChild *>(subList.at(subSize-1)->widget());
119  GLMdiChild *child2 = qobject_cast<GLMdiChild *>(subList.at(subSize-2)->widget());
120  StlCompare compare;
121 
122  char res = compare.init (child1->m_stlFile, child2->m_stlFile);
123  if (res != -1) {
124  char buf[4];
125  _itoa(res, buf, 10);
126  QMessageBox msgBox;
127  QRect geo = msgBox.geometry();
128  msgBox.resize (geo.width () + 40, geo.height () + 20);
129  msgBox.setText("Result of the Huffman Code.");
130 
131  QString str = "first file: ";
132  str += child1->m_curFile ;
133  str += "\n";
134  str += "second file: ";
135  str += child2->m_curFile;
136  str += "\n\n";
137  str += "The percent equality of two mesh is ";
138  str += buf;
139  str += "%";
140  msgBox.setInformativeText(str);
141  msgBox.exec();
142  }
143 
144  }
145 }
