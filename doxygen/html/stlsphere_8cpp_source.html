407 // Purpose:
408 // Cartesian to sphere coordinate transform.
409 // Discussion:
410 // We allow the trivial case M = 1; in that case alone, the value R must be assumed to have a sign.
411 // Parameters:
412 // Input, int M, the spatial dimension.
413 // 1 <= M.
414 // Input, int N, the number of points to transform.
415 // Input, double C[M], the center of the sphere.
416 // Input, double X[M*N], the Cartesian coordinates of the points.
417 // Output, double R[N], the radius of the points on the sphere.
418 // Except for the trivial case M = 1, R is assumed nonnegative.
419 // Output, double THETA[(M-1)*N], the coordinate angles of the points, measured in radians.
420 void StlSphere::cartesianToSphere(int m, int n, double c[], double x[],
421  double r[], double theta[] )
422 {
423  int i;
424  int i1;
425  int j;
426  double t;
427  double *top;
428  double *x2;
429 //
430 // Handle special case of M = 1.
431 //
432  if ( m == 1 )
433  {
434  for ( j = 0; j < n; j++ )
435  {
436  r[j] = x[0+j*m] - c[0];
437  }
438  return;
439  }
440 //
441 // Subtract the center.
442 //
443  x2 = new double[ m * n ];
444  for ( j = 0; j < n; j++ )
445  {
446  for ( i = 0; i < m; i++ )
447  {
448  x2[i+j*m] = x[i+j*m] - c[i];
449  }
450  }
451 //
452 // Compute R.
453 //
454  for ( j = 0; j < n; j++ )
455  {
456  t = 0.0;
457  for ( i = 0; i < m; i++ )
458  {
459  t = t + pow ( x2[i+m*j], 2 );
460  }
461  r[j] = sqrt ( t );
462  }
463 //
464 // Compute M-2 components of THETA.
465 //
466  for ( j = 0; j < n; j++ )
467  {
468  for ( i = 0; i < m - 1; i++ )
469  {
470  theta[i+j*(m-1)] = 0.0;
471  }
472  }
473 
474  for ( i = 1; i < m - 1; i++ )
475  {
476  for ( i1 = 0; i1 <= i - 1; i1++ )
477  {
478  for ( j = 0; j < n; j++ )
479  {
480  theta[i1+j*(m-1)] = theta[i1+j*(m-1)] + pow ( x2[i+j*m], 2 );
481  }
482  }
483  }
484 
485  for ( j = 0; j < n; j++ )
486  {
487  for ( i = 0; i < m - 2; i++ )
488  {
489  theta[i+j*(m-1)] = theta[i+j*(m-1)] + pow ( x2[m-1+j*m], 2 );
490  }
491  }
492 
493  for ( i = 0; i < m - 2; i++ )
494  {
495  for ( j = 0; j < n; j++ )
496  {
497  theta[i+j*(m-1)] = atan2 ( sqrt ( theta[i+j*(m-1)] ), x2[i+j*m] );
498  }
499  }
500 //
501 // Compute last component of THETA.
502 //
503  top = new double[n];
504 
505  for ( j = 0; j < n; j++ )
506  {
507  top[j] = sqrt ( pow ( x2[m-1+j*m], 2 ) + pow ( x2[m-2+j*m], 2 ) ) + x2[m-2+j*m];
508  }

512  theta[m-2+j*(m-1)] = 2.0 * atan2 ( x2[m-1+j*m], top[j] );
513  }
