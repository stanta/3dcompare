62 bool StlFile::initialize(const std::string& fileName)
63 {
64  m_stats.numFacets = 0;
65  m_stats.numPoints = 0;
66  m_stats.surface = -1.0;
67  m_stats.volume = -1.0;
68  // Open the file
69  m_file.open(fileName.c_str(), ::std::ios::binary);
70  if (m_file.is_open()) {
71  int numFacets;
72  // Find length of file
73  m_file.seekg(0, ::std::ios::end);
74  int fileSize = m_file.tellg();
75  // Check for binary or ASCII file
76  m_file.seekg(0, ::std::ios::beg);
77  m_stats.type = BINARY;
78  int c;
79  while((c = m_file.get()) != EOF && c <= 127)
80  ;
81  if(c == EOF) {
82  m_stats.type = ASCII;
83  }
84  m_file.seekg(0, ::std::ios::beg);
85  // Get the header and the number of facets in the .STL file
86  // If the .STL file is binary, then do the following
87  if (m_stats.type == BINARY) {
88  // Test if the STL file has the right size
89  if ((fileSize - HEADER_SIZE) % SIZE_OF_FACET != 0) {
90  ::std::cerr << "The file " << m_file << " has a wrong size."
91  << ::std::endl;
92  throw wrong_header_size();
93  }
94  numFacets = (fileSize - HEADER_SIZE) / SIZE_OF_FACET;
95  // Read the header
96  char buffer[JUNK_SIZE];
97  m_file.read(buffer, JUNK_SIZE);
98  m_stats.header = buffer;
99  // Read the int following the header.
100  // This should contain the number of facets
101  int headerNumFacets = readIntFromBytes(m_file);
102  if (numFacets != headerNumFacets) {
103  ::std::cerr << "Warning: File size doesn't match number of "
104  << "facets in the header." << ::std::endl;
105 #ifdef QT_CORE_LIB
106  QErrorMessage errMessage;
107  errMessage.showMessage("File size doesn't match number of facets "
108  "in the header.");
109  QApplication::restoreOverrideCursor();
110  errMessage.exec();
111  QApplication::setOverrideCursor(Qt::WaitCursor);
112 #endif
113  return false;
114  }
115  }
116  else { // Otherwise, if the .STL file is ASCII, then do the following
117  m_file.seekg(0, ::std::ios::beg);
118  // Get the header
119  getline(m_file, m_stats.header);
120  // Find the number of facets
121  int numLines = 0;
122  ::std::string line;
123  while (!getline(m_file, line).eof()) {
124  if (line.size() > 4) { // don't count short lines
125  numLines++;
126  }
127  }
128  m_file.seekg(0, ::std::ios::beg);
129  numFacets = numLines / ASCII_LINES_PER_FACET;
130  }
131  m_stats.numFacets += numFacets;
132  } else {
133  ::std::cerr << "The file " << m_file << " could not be found." << ::std::endl;
134  throw error_opening_file();
135  return false;
136  }
137  return true;
138 }

152 void StlFile::readData(int firstFacet, int first)
153 {
154  if (m_stats.type == BINARY) {
155  m_file.seekg(HEADER_SIZE, ::std::ios::beg);
156  } else {
157  m_file.seekg(0, ::std::ios::beg);
158  ::std::string line;
159  // Skip the first line of the file
160  getline(m_file, line);
161  }
162  Facet facet;
163  for (int i = firstFacet; i < m_stats.numFacets; i++)
164  {
165  if (m_stats.type == BINARY) { // Read a single facet from a binary .STL file
166  facet.normal.x = readFloatFromBytes(m_file);
167  facet.normal.y = readFloatFromBytes(m_file);
168  facet.normal.z = readFloatFromBytes(m_file);
169  facet.vector[0].x = readFloatFromBytes(m_file);
170  facet.vector[0].y = readFloatFromBytes(m_file);
171  facet.vector[0].z = readFloatFromBytes(m_file);
172  facet.vector[1].x = readFloatFromBytes(m_file);
173  facet.vector[1].y = readFloatFromBytes(m_file);
174  facet.vector[1].z = readFloatFromBytes(m_file);
175  facet.vector[2].x = readFloatFromBytes(m_file);
176  facet.vector[2].y = readFloatFromBytes(m_file);
177  facet.vector[2].z = readFloatFromBytes(m_file);
178  facet.extra[0] = m_file.get();
179  facet.extra[1] = m_file.get();
180  } else { // Read a single facet from an ASCII .STL file
181  ::std::string junk;
182  m_file >> junk >> junk;
183  m_file >> facet.normal.x >> facet.normal.y >> facet.normal.z;
184  m_file >> junk >> junk >> junk;
185  m_file >> facet.vector[0].x >> facet.vector[0].y >> facet.vector[0].z;
186  m_file >> junk;
187  m_file >> facet.vector[1].x >> facet.vector[1].y >> facet.vector[1].z;
188  m_file >> junk;
189  m_file >> facet.vector[2].x >> facet.vector[2].y >> facet.vector[2].z;
190  m_file >> junk >> junk;
191  }
192  // Write the facet into memory.
193  m_vFacets[i] = facet;
194 
195  // While we are going through all of the facets, let's find the
196  // maximum and minimum values for x, y, and z
197  // Initialize the max and min values the first time through
198  if (first) {
199  m_stats.max.x = facet.vector[0].x;
200  m_stats.min.x = facet.vector[0].x;
201  m_stats.max.y = facet.vector[0].y;
202  m_stats.min.y = facet.vector[0].y;
203  m_stats.max.z = facet.vector[0].z;
204  m_stats.min.z = facet.vector[0].z;
205 
206  float xDiff = std::abs(facet.vector[0].x - facet.vector[1].x);
207  float yDiff = std::abs(facet.vector[0].y - facet.vector[1].y);
208  float zDiff = std::abs(facet.vector[0].z - facet.vector[1].z);
209  float maxDiff = std::max(xDiff, yDiff);
210  maxDiff = std::max(zDiff, maxDiff);
211  m_stats.shortestEdge = maxDiff;
212  first = 0;
213  }
214  // Now find the max and min values
215  m_stats.max.x = std::max(m_stats.max.x, facet.vector[0].x);
216  m_stats.min.x = std::min(m_stats.min.x, facet.vector[0].x);
217  m_stats.max.y = std::max(m_stats.max.y, facet.vector[0].y);
218  m_stats.min.y = std::min(m_stats.min.y, facet.vector[0].y);
219  m_stats.max.z = std::max(m_stats.max.z, facet.vector[0].z);
220  m_stats.min.z = std::min(m_stats.min.z, facet.vector[0].z);
221 
222  m_stats.max.x = std::max(m_stats.max.x, facet.vector[1].x);
223  m_stats.min.x = std::min(m_stats.min.x, facet.vector[1].x);
224  m_stats.max.y = std::max(m_stats.max.y, facet.vector[1].y);
225  m_stats.min.y = std::min(m_stats.min.y, facet.vector[1].y);
226  m_stats.max.z = std::max(m_stats.max.z, facet.vector[1].z);
227  m_stats.min.z = std::min(m_stats.min.z, facet.vector[1].z);
228 
229  m_stats.max.x = std::max(m_stats.max.x, facet.vector[2].x);
230  m_stats.min.x = std::min(m_stats.min.x, facet.vector[2].x);
231  m_stats.max.y = std::max(m_stats.max.y, facet.vector[2].y);
232  m_stats.min.y = std::min(m_stats.min.y, facet.vector[2].y);
233  m_stats.max.z = std::max(m_stats.max.z, facet.vector[2].z);
234  m_stats.min.z = std::min(m_stats.min.z, facet.vector[2].z);
235  }
236  m_stats.size.x = m_stats.max.x - m_stats.min.x;
237  m_stats.size.y = m_stats.max.y - m_stats.min.y;
238  m_stats.size.z = m_stats.max.z - m_stats.min.z;
239  m_stats.boundingDiameter = sqrt(m_stats.size.x * m_stats.size.x +
240  m_stats.size.y * m_stats.size.y +
241  m_stats.size.z * m_stats.size.z);
242  m_stats.numPoints = getNumPoints();
243  //m_stats.surface = getSurface();
244  //m_stats.volume = getVolume();
245  // Set unit Sphere data(cennter of mass ...)
246  m_stlSphere->setUnitSphere(&m_vFacets, &m_stats);
247 }
