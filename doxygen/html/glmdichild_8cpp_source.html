133 bool GLMdiChild::saveAs()
134 {
135  if (!m_isUntitled) {
136  QString filterBin = tr("STL Files, binary (*.stl)");
137  QString filterAscii = tr("STL Files, ASCII (*.stl)");
138  QString filterAll = tr("All files (*.*)");
139  QString filterSel;
140  // Set the current file type as default
141  if (m_stlFile->getStats().type == ASCII)
142  filterSel = filterAscii;
143  else
144  filterSel = filterBin;
145  QString fileName = QFileDialog::getSaveFileName(this, tr("Save As"), m_curFile,
146  filterBin + ";;" + filterAscii + ";;" + filterAll, &filterSel);
147  if (fileName.isEmpty())
148  return false;
149  // Change the current file type to the one chosen by the user
150  if (filterSel == filterBin)
151  m_stlFile->setFormat(BINARY);
152  else if (filterSel == filterAscii)
153  m_stlFile->setFormat(ASCII);
154  // Save the file
155  return saveFile(fileName);
156  }
157  return false;
158 }
